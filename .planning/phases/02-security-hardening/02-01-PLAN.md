<!-- ENCRYPTION_KEY 格式规范 -->
- 64 字符 hex 字符串（= 32 字节 = 256-bit AES key）
- 生成方式：node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"
- 存储于 .env 文件，通过 process.env.ENCRYPTION_KEY 访问

<!-- encrypt(plaintext) 输出格式 -->
"enc:v1:<base64url(iv[12] + ciphertext[N] + tag[16])>"
- iv: 12 字节随机（AES-GCM 推荐 nonce 长度）
- tag: GCM 认证标签 16 字节
- base64url 避免 '+' '/' 字符，方便存入 SQLite TEXT 列

<!-- decrypt(ciphertext) 行为 -->
- 若输入以 "enc:v1:" 开头：解密并返回明文
- 若输入不以 "enc:v1:" 开头（或为空/null）：原样返回（兼容过渡期明文值）
- 解密失败时抛出错误（不静默失败）

  <name>Task 1: 创建 services/crypto.js</name>
  <files>services/crypto.js</files>
  <action>
    创建新文件 `services/crypto.js`，内容如下：
    
    ```js
    const crypto = require('crypto');
    
    const ALGO = 'aes-256-gcm';
    const IV_BYTES = 12;
    const TAG_BYTES = 16;
    const PREFIX = 'enc:v1:';
    
    function getKey() {
      const hex = process.env.ENCRYPTION_KEY || '';
      if (hex.length !== 64 || !/^[0-9a-f]+$/i.test(hex)) {
        throw new Error('ENCRYPTION_KEY 未设置或格式错误（需 64 位 hex）');
      }
      return Buffer.from(hex, 'hex');
    }
    
    function encrypt(plaintext) {
      if (!plaintext) return plaintext;
      const key = getKey();
      const iv = crypto.randomBytes(IV_BYTES);
      const cipher = crypto.createCipheriv(ALGO, key, iv);
      const enc = Buffer.concat([cipher.update(String(plaintext), 'utf8'), cipher.final()]);
      const tag = cipher.getAuthTag();
      const payload = Buffer.concat([iv, enc, tag]);
      return PREFIX + payload.toString('base64url');
    }
    
    function decrypt(ciphertext) {
      if (!ciphertext || !String(ciphertext).startsWith(PREFIX)) return ciphertext;
      const key = getKey();
      const payload = Buffer.from(String(ciphertext).slice(PREFIX.length), 'base64url');
      const iv = payload.subarray(0, IV_BYTES);
      const tag = payload.subarray(payload.length - TAG_BYTES);
      const enc = payload.subarray(IV_BYTES, payload.length - TAG_BYTES);
      const decipher = crypto.createDecipheriv(ALGO, key, iv);
      decipher.setAuthTag(tag);
      return decipher.update(enc) + decipher.final('utf8');
    }
    
    module.exports = { encrypt, decrypt };
    ```
    
    **设计要点：**
    - `getKey()` 每次调用时读取 env var，不在模块加载时缓存（允许测试环境动态设置）
    - `encrypt()` 对空值原样返回（避免加密 null/undefined/''）
    - `decrypt()` 对无前缀值原样返回（明文过渡期兼容）
    - AES-GCM 提供认证加密，篡改数据会在 decipher.final() 时抛出错误

    在其**之后**追加 ENCRYPTION_KEY 验证：
    ```js
    const _encKey = process.env.ENCRYPTION_KEY || '';
    if (!_encKey || _encKey.length !== 64 || !/^[0-9a-f]+$/i.test(_encKey)) {
      if (isProduction) {
        throw new Error('ENCRYPTION_KEY is required in production (generate: node -e "console.log(require(\'crypto\').randomBytes(32).toString(\'hex\'))")');
      } else if (_encKey) {
        console.warn('[security] ENCRYPTION_KEY 格式错误（需 64 位 hex）—— 在开发环境中忽略，生产环境将启动失败');

    }
    ```

    **设计说明：**
    - 生产环境（`NODE_ENV=production`）：key 缺失或格式错误时直接抛出，阻止启动

    - 不在 server.js 中 `require('./services/crypto')`，避免模块加载顺序问题
