  创建新文件 `utils/shell-escape.js`，内容如下：

  ```js
  function dqEsc(s) {
    return String(s)
      .replace(/\\/g, '\\\\')
      .replace(/`/g, '\\`')
      .replace(/\$/g, '\\$')
      .replace(/"/g, '\\"');
  }

  function shSingleQuote(s) {
    return `'${String(s).replace(/'/g, `'\\''`)}'`;
  }

  module.exports = { dqEsc, shSingleQuote };
  ```
  </action>
